pub async fn test_mirror_speed(url: &str) -> Result<u64> {
    let client = reqwest::Client::builder()
        .timeout(Duration::from_secs(5))
        .build()?;

    let start = Instant::now();
    let response = client.head(url).send().await?;

    if response.status().is_success() {
        let latency = start.elapsed().as_millis() as u64;
        Ok(latency)
    } else {
        Err(Error::MirrorUnavailable)
    }
}

pub async fn benchmark_mirrors(
    mirrors: Vec<MirrorSource>
) -> Vec<SpeedTestResult> {
    let tasks: Vec<_> = mirrors
        .into_iter()
        .map(|mirror| {
            let mirror_clone = mirror.clone();
            async move {
                let result = test_mirror_speed(&mirror_clone.url).await;
                SpeedTestResult {
                    mirror: mirror_clone,
                    latency_ms: result.unwrap_or(u64::MAX),
                    success: result.is_ok(),
                }
            }
        })
        .collect();

    let results = futures::future::join_all(tasks).await;
    results.into_iter().collect()
}

impl EnvManager {
    pub fn get_user_env_vars(&self) -> Result<HashMap<String, String>> {
        let content = fs::read_to_string(&self.config_file)?;
        let vars = content
            .lines()
            .filter_map(|line| {
                let line = line.trim();
                if line.starts_with("export ") {
                    let line = line.strip_prefix("export ")?;
                    let mut parts = line.splitn(2, '=');
                    Some((
                        parts.next()?.to_string(),
                        parts.next()?.trim_matches('"').to_string(),
                    ))
                } else {
                    None
                }
            })
            .collect();

        Ok(vars)
    }

    pub fn set_env_var(&self, key: &str, value: &str) -> Result<()> {
        let mut content = fs::read_to_string(&self.config_file)
            .unwrap_or_default();

        // æ£€æŸ¥æ˜¯å¦å·²å­˜åœ¨

                r"^export {}=.*$",

        }
